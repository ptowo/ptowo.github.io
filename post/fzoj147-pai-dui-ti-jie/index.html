FZOJ#147. 排队 题解

2022-05-19 动态规划

抢饭是高中生活的一部分，现在有一列队伍长度为 nnn，(注意：由于人与人之间要保持距离，且不同情况所保持的距离大小不同，所以长度并不能直接体现队列的人数)。已知男男之间的距离为 aaa ，男女之间距离为 bbb ，女女之间距离为 ccc。一个男生打饭时间为 ddd ，一个女生打饭时间为 eee ，求所有情况的排队时间总和（忽略身体的大小对队伍长度的贡献），答案对 109+710^9+7109+7 取模。

Data Range:1≤n≤1018,1≤a,b,c,d,e≤30,a,b,c≤n\text{Data Range:} 1 \leq n \leq 10^18, 1\leq a,b,c,d,e \leq 30, a,b,c \leq nData Range:1≤n≤1018,1≤a,b,c,d,e≤30,a,b,c≤n。

一眼 dp，dpi,opdp_{i, op}dpi,op​ 表示当前在第 iii 个位置放了一个 opopop 类型人之后的用时，同时记录每种情况的方案数 gi,opg_{i,op}gi,op​。

然后暴力枚举情况转移可以得到 50 分，显然矩阵加速，写个矩阵就完了。

矩阵构造将答案矩阵放在一行，这样才能很方便的转移，预处理出前 404040 个 dp 值。

注意对于 ggg 的构造，要整体后移一个，因为对应推到 i+1i + 1i+1 式子中时需要的是 gi+1,opg_{i+1,op}gi+1,op​。

详细见代码：

// 德丽莎你好可爱德丽莎你好可爱德丽莎你好可爱德丽莎你好可爱德丽莎你好可爱
// 德丽莎的可爱在于德丽莎很可爱，德丽莎为什么很可爱呢，这是因为德丽莎很可爱！
#include <bits/stdc++.h>
#define int long long
using namespace std;
inline int read() {
  int x = 0, f = 1;  char ch = getchar();
  while( !isdigit(ch) ) { if(ch == '-') f = -1;  ch = getchar();  }
  while( isdigit(ch) ) {  x = (x << 1) + (x << 3) + (ch ^ 48);  ch = getchar();  }
  return x * f;
}
const int N = 325, mod = 1e9 + 7;
int n, a, b, c, d, e, lim = 40, dp[N][3], g[N][3];
struct mt {
  int a[250][250];
  mt() { memset(a, 0, sizeof(a)); }
  mt operator * (const mt &b) {
    mt res;
    for (int i = 1; i <= lim * 4; i++) 
      for (int k = 1; k <= lim * 4; k++) 
        for (int j = 1; j <= lim * 4; j++) 
          res.a[i][j] += a[i][k] * b.a[k][j] % mod, res.a[i][j] %= mod;
    return res;
  }
}base, ans;
void ksm(mt a,int k) {
  while (k) { if (k & 1) ans = ans * a; k >>= 1; a = a * a; }
}
int id(int x) { return max(x, 0ll); }
signed main () {
  n = read(), a = read(), b = read(), c = read(), d = read(), e = read();
  dp[0][0] = d; g[0][0] = 1;
  dp[0][1] = e; g[0][1] = 1;
  for (int i = 1; i <= lim + 1; i++) {
    if (i - a >= 0) g[i][0] += g[i - a][0], dp[i][0] += dp[i - a][0];
    if (i - c >= 0) g[i][1] += g[i - c][1], dp[i][1] += dp[i - c][1];
    if (i - b >= 0) g[i][1] += g[i - b][0], g[i][0] += g[i - b][1], dp[i][0] += dp[i - b][1], dp[i][1] += dp[i - b][0]; 
    dp[i][0] += g[i][0] * d; dp[i][1] += g[i][1] * e;
    dp[i][0] %= mod, dp[i][1] %= mod; g[i][0] %= mod; g[i][1] %= mod;
  }
  for (int i = 1; i <= lim; i++) ans.a[1][i] = dp[i][0];
  for (int i = lim + 1; i <= lim + lim; i++) ans.a[1][i] = dp[i - lim][1];
  for (int i = 2 * lim + 1; i <= 3 * lim; i++) ans.a[1][i] = g[i - lim * 2 + 1][0];
  for (int i = 3 * lim + 1; i <= 4 * lim; i++) ans.a[1][i] = g[i - lim * 3 + 1][1];
  for (int i = 1; i < lim; i++) base.a[i + 1][i] = 1;
  for (int i = lim + 1; i < 2 * lim; i++) base.a[i + 1][i] = 1;
  for (int i = lim * 2 + 1; i < 3 * lim; i++) base.a[i + 1][i] = 1;
  for (int i = lim * 3 + 1; i < 4 * lim; i++) base.a[i + 1][i] = 1;
  //转移 f[i][0]
  base.a[lim + 1 - a][lim] = 1;
  base.a[lim * 2 + 1 - b][lim] = 1;
  base.a[lim * 3][lim] = d;
  //lim -> lim + 1
  // a[1][lim] a[1][k] * base[k][lim]
  
  //转移 f[i][1]
  base.a[lim * 2 + 1 - c][lim * 2] = 1;
  base.a[lim + 1 - b][lim * 2] = 1;
  base.a[lim * 4][lim * 2] = e;
  // lim * 2 - >lim * 2 + 1 
  
  //转移 g[i][0]
  base.a[lim * 3 + 1 - a][lim * 3] = 1;
  base.a[lim * 4 + 1 - b][lim * 3] = 1;
  // lim*3 -> lim * 3 + 1

  //转移 g[i][1]
  base.a[lim * 4 + 1 - c][lim * 4] = 1;
  base.a[lim * 3 + 1 - b][lim * 4] = 1;
  if (n <= lim) {
    int ans1 = dp[n][0] + dp[n][1]; ans1 %= mod; ans1 += mod; ans1 %= mod;
    printf("%lld\n", ans1);
  }  else {
    int qwq = n - lim;
    ksm(base, qwq);
    int ans1 = ans.a[1][lim] + ans.a[1][2 * lim];
    ans1 %= mod;
    cout << ans1 << "\n";
  }
  //lim * 4 -> lim * 4 + 1
  // f[i][0] = f[i - 1][0] + f[i - 2][1]
  // f[5][0] = f[4][0] + f[2][1] + 2 
  return 0;
}
